cmake_minimum_required(VERSION 3.16)
project(DEADGL VERSION 0.3.0 LANGUAGES C)
set(CMAKE_C_STANDARD 99)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_library(deadgl STATIC src/deadgl.c)
target_include_directories(deadgl PUBLIC include)
if (UNIX)
    target_link_libraries(deadgl PUBLIC m)
endif()
add_executable(deadgl_cli src/deadgl_cli.c)
target_link_libraries(deadgl_cli PRIVATE deadgl)
set_target_properties(deadgl_cli PROPERTIES OUTPUT_NAME deadgl)
add_executable(test_deadgl tests/test_deadgl.c)
target_link_libraries(test_deadgl PRIVATE deadgl)
if (MSVC)
    target_compile_options(deadgl PRIVATE /W4 /WX)
    target_compile_options(deadgl_cli PRIVATE /W4 /WX)
    target_compile_options(test_deadgl PRIVATE /W4 /WX)
else()
    target_compile_options(deadgl PRIVATE -Wall -Wextra -Wpedantic -Wshadow -Wstrict-prototypes -Werror)
    target_compile_options(deadgl_cli PRIVATE -Wall -Wextra -Wpedantic -Wshadow -Wstrict-prototypes -Werror)
    target_compile_options(test_deadgl PRIVATE -Wall -Wextra -Wpedantic -Wshadow -Wstrict-prototypes -Werror)
endif()
enable_testing()
add_test(NAME core COMMAND test_deadgl)
add_test(NAME version COMMAND deadgl_cli --version)
add_test(NAME demo COMMAND deadgl_cli demo shrine -o shrine.ppm)
